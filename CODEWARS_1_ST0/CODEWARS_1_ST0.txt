const quarterOf = (month) => {
  if (month >= 1 && month <= 3 ){
    return 1;
  } else if ( month >= 4 && month <= 6 ){
    return 2;

C3+++

https://www.codewars.com/kata/capitalization-and-mutability

Your coworker was supposed to write a simple helper function to capitalize a string (that contains a single word) before they went on vacation.

Unfortunately, they have now left and the code they gave you doesn't work. Fix the helper function they wrote so that it works as intended (i.e. it must make the first character in the string upper case).
The string will always start with a letter and will never be empty.

Examples:
"hello" --> "Hello"
"Hello" --> "Hello" (the first letter was already capitalized)
"a"     --> "A"

РЕШЕНИЕ:

function capitalizeWord(word) {
  const wordCapitalize = word[0].toUpperCase() + word.slice(1);
  return wordCapitalize;
}

"use strict"

let targetString = "45385593107843568";

function fakeBin(targetString) {
  
  let iterableTarget = targetString.split("");
  let arrayCharacters = [];
  
  for( let char of iterableTarget ){

    if( char < 5 ){
      arrayCharacters.unshift(0);  
    } else {
      arrayCharacters.unshift(1); 
    }
  }
  return arrayCharacters.join("");
}

https://www.codewars.com/kata/largest-square-inside-a-circle

Determine the area of the largest square that can fit inside a circle with radius r.

РЕШЕНИЕ:

Если у нас есть радиус описанной окружности
Возведем его в квадрат и умножим на два.
S = 2 × R * R, где R — это радиус описанной окружности. 
Источник - Онлайн школа Skysmart: https://skysmart.ru/articles/mathematic/ploshad-kvadrata

function areaLargestSquare(r) {
  return 2 * ( r * r ) ;
}

https://www.codewars.com/kata/simple-multiplication

This kata is about multiplying a given number by eight if it is an even number and by nine otherwise.

РЕШЕНИЕ:

function simpleMultiplication(number) {
  if ( number % 2 === 0 ){
    return number * 8;
  }
  return number * 9;
}
